% ── Slide 6 · I-PPO Algorithm ────────────────────────────────────────────────
\begin{frame}{Independent PPO (I-PPO) --- How Each Agent Learns}
\begin{columns}[T]
\begin{column}{0.50\textwidth}
\vspace{0.1cm}
\begin{tcolorbox}[mybox, title={PPO Clipped Objective}]
\vspace{-0.1cm}
\[
  L^{\text{CLIP}} = \mathbb{E}\!\left[\min\!\left(
    r_t \hat{A}_t,\;
    \mathrm{clip}(r_t,\,1{-}\epsilon,\,1{+}\epsilon)\hat{A}_t
  \right)\right]
\]
\vspace{-0.2cm}
\small where $r_t = \dfrac{\pi_\theta(a_t|s_t)}{\pi_{\theta_\mathrm{old}}(a_t|s_t)}$
\end{tcolorbox}

\vspace{0.2cm}
\begin{tcolorbox}[accentbox]
\small
\textbf{Agent-Specific Reward Shaping:}\\[2pt]
\footnotesize\textbf{ScaleOut:} under/over-provisioning + running cost + proactive bonus\\[1pt]
\textbf{Consolidation:} cost saving + SLA bonus $-$ migration penalty\\[1pt]
\textbf{Scheduling:} $-$\,mean wait time $-$ SLA violations
\end{tcolorbox}
\end{column}

\begin{column}{0.48\textwidth}
% Training loop diagram
\begin{tikzpicture}[
  sbox/.style={rectangle, rounded corners=3pt, draw=ACprimary, fill=ACprimary!10,

\begin{column}{0.47\textwidth}
\vspace{0.2cm}
\begin{tcolorbox}[mybox, title={Why MC Dropout?}]
\small
Keep dropout \textbf{on} at inference.\\
Run 30 forward passes $\to$ 30 different predictions.\\[4pt]
$\sigma = \mathrm{var}(\text{30 outputs})$\\[4pt]
\textbf{Low} $\sigma$ $\to$ model confident $\to$ act normally\\
\textbf{High} $\sigma$ $\to$ model uncertain $\to$ Safety Coordinator blocks drains and forces scale-up
\end{tcolorbox}

\vspace{0.2cm}
\begin{tcolorbox}[accentbox]
\small \textbf{Forecaster output injected into obs[200:215]:}\\
means $[\hat{d}_{t+1}, \hat{d}_{t+5}, \hat{d}_{t+10}, \hat{d}_{t+15}]$\\
sigmas $[\sigma_{t+1}, \sigma_{t+5}, \sigma_{t+10}, \sigma_{t+15}]$
\end{tcolorbox}

\vspace{0.2cm}
\begin{tcolorbox}[warnbox]
\small \textbf{Cost:} 1 model $\times$ 30 passes vs 5 ensemble models $\to$ \textbf{5$\times$ cheaper}
\end{tcolorbox}
\end{column}

\textbf{Formal Problem (Dec-POMDP)}
\vspace{0.15cm}

Node features (CPU, mem, status), job features (wait, priority), global metrics + forecaster $\hat{d},\sigma$\\[4pt]
\textbf{Actions} (decomposed via I-PPO):\\
$\cdot$ ScaleOut: $\{0,+1,+2\}$ VMs\\
$\cdot$ Consolidation: drain mask $\{0,1\}^{20}$\\
$\cdot$ Scheduling: priority strategy $\{0..4\}$\\[4pt]

\begin{column}{0.45\textwidth}
\vspace{0.1cm}
\begin{tcolorbox}[mybox, title={Why I-PPO wins}]
\footnotesize
\textbf{1. Smaller action spaces}\\
\scriptsize Each agent: 3, $2^{20}$, or 5 --- not 15M joint\\[2pt]
\footnotesize \textbf{2. Temporal hierarchy}\\
\scriptsize Fast decisions (30s) don't mix with slow (5min)\\[2pt]
\footnotesize \textbf{3. Per-agent rewards}\\
\scriptsize Clearer learning signal per subtask\\[2pt]
\footnotesize \textbf{4. Credit assignment}\\
\scriptsize Easier to debug --- which agent caused a violation?
\end{tcolorbox}

\begin{tcolorbox}[warnbox]

% ── Slide 14 · Limitations & Future Work ─────────────────────────────────────
\begin{frame}{Limitations \& Future Directions}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{tcolorbox}[mybox, title={Current Limitations}]
\small
\textbf{1. Homogeneous VMs}\\
\footnotesize Real clouds offer heterogeneous instances with varying CPU/memory/price\\[4pt]
\small \textbf{2. Single-cluster scope}\\
\footnotesize Production systems span thousands of clusters across regions\\[4pt]
\small \textbf{3. Simplified networking}\\
\footnotesize No cross-rack latency or bandwidth modelling\\[4pt]
\small \textbf{4. Single-day training}\\
\footnotesize Multi-day training with domain randomisation would improve generalisation
\end{tcolorbox}
\end{column}
\begin{column}{0.48\textwidth}
\vspace{0.0cm}
\begin{tcolorbox}[accentbox]
\small \textbf{Future Directions:}\\[3pt]
\footnotesize
\textbullet{} Heterogeneous VMs with bin-packing strategies\\[2pt]
\textbullet{} Multi-cluster federation (hierarchical agents)\\[2pt]
\textbullet{} Vertical scaling (resize individual VMs)\\[2pt]
\textbullet{} Transfer learning across traces (Azure, Borg)\\[2pt]
\textbullet{} Spot/preemptible instance price-aware scaling\\[2pt]
\textbullet{} Real Kubernetes deployment
\end{tcolorbox}
\vspace{0.15cm}
\begin{tcolorbox}[warnbox]
\small \textbf{Key open question:}\\
\footnotesize Can temporal I-PPO decomposition scale to 100+ node types and multi-region federated clusters?
\end{tcolorbox}
\end{column}
\end{columns}
\end{frame}
